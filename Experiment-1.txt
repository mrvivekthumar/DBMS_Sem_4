CREATE DATABASE IF NOT EXISTS EXPERIMENT_1;

USE EXPERIMENT_1;

CREATE TABLE SALESMAN(
	SID VARCHAR(5) PRIMARY KEY,
    SNAME VARCHAR(30) NOT NULL,
    CITY VARCHAR(15) NOT NULL,
    SALARY NUMERIC(5,2),
    PRODUCT VARCHAR(20),
    TGTTOGET NUMERIC(5,2),
    COMM NUMERIC(5,2)
);

CREATE TABLE CUSTOMERS(
	CID VARCHAR(5) PRIMARY KEY,
    CNAME VARCHAR(20) NOT NULL,
    CITY VARCHAR(15) NOT NULL,
    STATE VARCHAR(15),
    PINCODE NUMERIC(8),
    PRODUCT VARCHAR(20),
    CLASS CHAR(1) DEFAULT 'A'
);

CREATE TABLE ORDERS(
	OID VARCHAR(5) PRIMARY KEY,
    CID VARCHAR(5),
    SID VARCHAR(5),
    PRODUCT VARCHAR(20),
    QTY NUMERIC(5),
    ODATE DATE NOT NULL,
    O_AMT NUMERIC(8,2),
    FOREIGN KEY (CID) REFERENCES CUSTOMERS(CID),
    FOREIGN KEY (SID) REFERENCES SALESMAN(SID)
);

SHOW TABLES;
SELECT * FROM CUSTOMERS;
