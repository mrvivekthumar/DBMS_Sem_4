SELECT ORDERS.CID, CNAME, CITY, O_AMT
FROM CUSTOMERS
    JOIN ORDERS ON CUSTOMERS.CID = ORDERS.CID;

SELECT * FROM CUSTOMERS JOIN ORDERS ON CUSTOMERS.CID = ORDERS.CID;

SELECT *
FROM CUSTOMERS
    JOIN ORDERS ON CUSTOMERS.CID = ORDERS.CID
WHERE
    ORDERS.SID = 'S105'
    OR ORDERS.SID = 'S102';

SELECT SALESMAN.*, CUSTOMERS.CNAME
FROM
    SALESMAN
    JOIN ORDERS ON SALESMAN.SID = ORDERS.SID
    JOIN CUSTOMERS ON ORDERS.CID = CUSTOMERS.CID;
