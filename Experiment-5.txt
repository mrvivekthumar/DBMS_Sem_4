-- 1
SELECT ORDERS.*, SALESMAN.SNAME
FROM ORDERS
    INNER JOIN SALESMAN ON SALESMAN.sid = ORDERS.sid
WHERE
    EXTRACT(
        DAY
        FROM ODATE
    ) <= 10;

-- 2
SELECT SNAME, MIN(O_AMT)
FROM ORDERS
    INNER JOIN SALESMAN ON SALESMAN.SID = ORDERS.CID
GROUP BY
    SNAME;

-- 3
SELECT SID, SUM(O_AMT) AS TOTAL FROM ORDERS GROUP BY SID;
